PROJECT(k3dsdk-qtui-script)

FILE(GLOB HEADERS *.h)
FILE(GLOB SOURCES *.cpp)

LIST(SORT HEADERS)
LIST(SORT SOURCES)

SET(MOC_BUILT_SOURCES )
QT4_WRAP_CPP(MOC_BUILT_SOURCES
	)

ADD_DEFINITIONS(-DQT_NO_KEYWORDS)

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})
INCLUDE_DIRECTORIES(${QT_INCLUDES})

K3D_ADD_LIBRARY(k3dsdk-qtui-script SHARED ${HEADERS} ${SOURCES} ${MOC_BUILT_SOURCES})
K3D_GENERATE_DEF_FILE(k3dsdk-qtui-script)

TARGET_LINK_LIBRARIES(k3dsdk-qtui-script k3dsdk-qtui)

INSTALL(TARGETS k3dsdk-qtui-script
	RUNTIME DESTINATION bin
	LIBRARY DESTINATION ${K3D_LIBDIR}
	ARCHIVE DESTINATION ${K3D_LIBDIR}
	)
