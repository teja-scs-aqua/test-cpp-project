PROJECT(k3dsdk-subdivision-surface)

FILE(GLOB HEADERS *.h)
FILE(GLOB SOURCES *.cpp)

INCLUDE_DIRECTORIES(${k3d_SOURCE_DIR})
INCLUDE_DIRECTORIES(${k3dsdk_BINARY_DIR})
INCLUDE_DIRECTORIES(${Boost_INCLUDE_DIRS})
INCLUDE_DIRECTORIES(${K3D_GLIBMM_INCLUDE_DIRS})

LINK_DIRECTORIES(${K3D_SIGC_LIB_DIRS})

K3D_ADD_LIBRARY(k3dsdk-subdivision-surface SHARED ${HEADERS} ${SOURCES})
K3D_GENERATE_DEF_FILE(k3dsdk-subdivision-surface)
TARGET_LINK_LIBRARIES(k3dsdk-subdivision-surface k3dsdk)

INSTALL(TARGETS k3dsdk-subdivision-surface
	RUNTIME DESTINATION bin 
	LIBRARY DESTINATION ${K3D_LIBDIR}
	ARCHIVE DESTINATION ${K3D_LIBDIR})
